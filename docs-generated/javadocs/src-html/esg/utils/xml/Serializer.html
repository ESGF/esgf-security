031       /** 
032        * Method to write a DOM document to file.
033        * NOTE: the XML will be re-formatted for pretty printing.
034        * @param root the root element of the DOM document
035        * @param outputFile the intended XML output file
036        * @exception IOException
037        */
038       public static void DOMtoFile(org.w3c.dom.Element root, String outputFile) 
039                          throws java.io.IOException {
040    
041          org.apache.xml.serialize.OutputFormat of
042              = new org.apache.xml.serialize.OutputFormat(org.apache.xml.serialize.Method.XML,"UTF-8",false);
043          org.apache.xml.serialize.XMLSerializer xs
044              = new org.apache.xml.serialize.XMLSerializer(of);
045          java.io.FileWriter writer = new java.io.FileWriter(outputFile);
046          xs.setOutputCharStream(writer);
047          xs.serialize(root);
048          writer.close();
049            
050       } // DOMtoFile
051       
052       public static String DOMtoString(org.w3c.dom.Element root) throws java.io.IOException {
053               
054              org.apache.xml.serialize.OutputFormat of
055              = new org.apache.xml.serialize.OutputFormat(org.apache.xml.serialize.Method.XML,"UTF-8",false);
056          org.apache.xml.serialize.XMLSerializer xs
057              = new org.apache.xml.serialize.XMLSerializer(of);
058          java.io.StringWriter writer = new StringWriter();
059          xs.setOutputCharStream(writer);
060          xs.serialize(root);
061          writer.close();
062          return writer.toString();
063               
064       }
065    
066       /** 
067        * Method to write a DOM document to System.out 
068        * @param root the root element of the DOM document
069        * @exception IOException
070        */
071       public static void DOMout(org.w3c.dom.Element root) throws java.io.IOException {
072    
073          org.apache.xml.serialize.OutputFormat of
074              = new org.apache.xml.serialize.OutputFormat(org.apache.xml.serialize.Method.XML,"UTF-8",true);
075          org.apache.xml.serialize.XMLSerializer xs
076              = new org.apache.xml.serialize.XMLSerializer(of);
077          xs.setOutputByteStream(System.out);
078          xs.serialize(root);
079    
080       } // DOMout

082       /**
083        * Method to write a JDOM document to file
084        * @param jdoc the JDOM document
085        * @param outputFile the intended xml output file
086        * @exception IOException
087        */
088       public static void JDOMtoFile(org.jdom.Document jdoc, String outputFile)
089                          throws java.io.IOException {
090    
091          org.jdom.output.XMLOutputter outputter = getXMLOutputter();
092          java.io.FileWriter writer = new java.io.FileWriter(outputFile);
093          outputter.output(jdoc, writer);
094          writer.close();
095    
096       } // JDOMtoFile()
097    
098       /** 
099        * Method to deserialize a JDOM document from a  String and write it to a file 
100        * @param xml the String containig the serialized xml document
101        * @param outputFile the intended xml output file
102        * @exception IOException
103        * @exception JDOMException
104        */
105       public static void JDOMtoFile(String xml, String outputFile)
106                          throws java.io.IOException, org.jdom.JDOMException {
107    
108          java.io.StringReader sr = new java.io.StringReader(xml);
109          org.jdom.input.SAXBuilder sb = new org.jdom.input.SAXBuilder();
110          org.jdom.Document jdoc =  sb.build(sr);
111          org.jdom.output.XMLOutputter outputter = getXMLOutputter();
112          java.io.FileWriter writer = new java.io.FileWriter(outputFile);
113          outputter.output(jdoc, writer);
114          writer.close();
115    
116       } // JDOMtoFile
117       
118       private static org.jdom.output.XMLOutputter getXMLOutputter() {
119            org.jdom.output.Format format = org.jdom.output.Format.getPrettyFormat();
120            format.setLineSeparator(System.getProperty("line.separator"));
121            org.jdom.output.XMLOutputter outputter = new org.jdom.output.XMLOutputter(format);
122            return outputter;
123       } // getXMLOutputter()

130       public static void JDOMout(org.jdom.Document jdoc) 
131                          throws java.io.IOException {
132    
133          org.jdom.output.XMLOutputter outputter = getXMLOutputter();
134          outputter.output(jdoc, System.out);
135    
136       } // JDOMout()

139        * Method to deserialize a JDOM document from a  String and write it to System.out

141        * @exception IOException
142        * @exception JDOMException
143        */
144       public static void JDOMout(String xml) 
145              throws java.io.IOException, org.jdom.JDOMException {
146    
147          java.io.StringReader sr = new java.io.StringReader(xml);

149          org.jdom.Document jdoc =  sb.build(sr);
150          org.jdom.output.XMLOutputter outputter = getXMLOutputter();
151          outputter.output(jdoc, System.out);
152    
153       } // JDOMtoFile

155      /**
156       * Method to serialize a JDOM document to a string representation
157       * @param jdoc : XML document as org.jdom.Document object
158       */
159      public static String JDOMtoString(org.jdom.Document jdoc) throws org.jdom.JDOMException {
160        return JDOMtoString(jdoc, false); // compact form
161      } // JDOMtoString()
162      
163      /**
164       * Method to serialize a JDOM document to a string representation
165       * @param jdoc : XML document as org.jdom.Document object
166       * @param pretty : true to output XML in pretty format
167       */
168      public static String JDOMtoString(org.jdom.Document jdoc, boolean pretty) throws org.jdom.JDOMException {
169            org.jdom.output.Format format = (pretty ? org.jdom.output.Format.getPrettyFormat() 
170                                                                : org.jdom.output.Format.getCompactFormat());
171              org.jdom.output.XMLOutputter outputter = new org.jdom.output.XMLOutputter(format);
172        return outputter.outputString(jdoc);
173      } // JDOMtoString()  
174

176       * Method to serialize a JDOM element to a string representation

179      public static String JDOMtoString(org.jdom.Element jelem) throws org.jdom.JDOMException {
180        //org.jdom.output.XMLOutputter outputter= new org.jdom.output.XMLOutputter(org.jdom.output.Format.getCompactFormat());
181        org.jdom.output.XMLOutputter outputter= new org.jdom.output.XMLOutputter();
182        return outputter.outputString(jelem);
183      } // JDOMtoString()

185      /** Method to serialize a javax.xml.transform.Source object to a String */
186      public static String SourceToString(javax.xml.transform.Source source) throws javax.xml.transform.TransformerException {
187    
188       javax.xml.transform.TransformerFactory tFactory = javax.xml.transform.TransformerFactory.newInstance();
189       javax.xml.transform.Transformer transformer = tFactory.newTransformer(); // copies source --> result
190       // execute transformation
191       java.io.CharArrayWriter caw = new java.io.CharArrayWriter();
192       javax.xml.transform.stream.StreamResult result = new javax.xml.transform.stream.StreamResult(caw);
193       transformer.transform(source, result);
194       return caw.toString();
195    
196      } // SourceToString()

199      public static String toHTML(String xml) throws Exception {
200                    return xml.replaceAll("&","&amp;").replaceAll("<","&lt;").replaceAll(">","&gt;");
