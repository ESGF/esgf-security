019    package esg.utils.xml;
020    
021    import java.io.IOException;
022    import java.io.InputStream;
023    import java.io.StringReader;
024    import java.util.Enumeration;
025    import java.util.PropertyResourceBundle;
026    import java.util.ResourceBundle;
027    
028    import org.jdom.Document;
029    import org.jdom.JDOMException;
030    import org.jdom.input.SAXBuilder;
031    import org.jdom.output.XMLOutputter;
032    
033    /**
034     * Utility class to parse input XML documents. It contains only static methods. 
035     */
036    public class Parser {
037            
038            // XML parser of choice
039            final static String parser = "org.apache.xerces.parsers.SAXParser";
040            static SAXBuilder builder = null;  // shared instance
041            
042      /** Main method
043       * 
044       * @param xmlfile: local patgh or URL of XML file
045       * @param validate: false to check XML well-formdness, true to perform schema validation
046       * @return
047       * @throws IOException
048       * @throws JDOMException
049       */
050            public static void main(String[] args) throws JDOMException, IOException {
051              
052                    // input arguments
053                    if (args.length!=2) {
054                            System.out.println("USAGE: ucar.xml.Parser xmlURI validate");
055                            System.exit(-1);
056                    }
057                    String xmlUri = args[0];
058                    boolean validate = (Boolean.valueOf(args[1]).booleanValue());
059                    
060                    init(validate);
061                    Document doc = toJDOM(xmlUri);
062                    
063            }
064                    
065            /** Class initialization method instantiates a validating or non-validating
066             *  parser instance that may be shared by all succesive following calls  
067             *  @param validate : true to enable schema validation ,false otherwise
068             */
069            public static synchronized void init(boolean validate) {
070                    if (builder==null) builder = getBuilder(validate);
071            } // init()
072    
073      /**
074       * Method to parse an input xml file and return a string representation
075       * NOTE: this method uses the shared parser instance initialized by a previous init() call
076       * @param xmlfile : local pathname or uri of XML file
077       */
078      public static String toString(String xmlfile) throws JDOMException, IOException {
079        Document jdoc = toJDOM(xmlfile);
080        XMLOutputter outputter = new XMLOutputter(org.jdom.output.Format.getCompactFormat());
081        return outputter.outputString(jdoc);
082      } // toString()
083    
084      /**
085       * Method to parse an input xml file and return a JDOM document
086       * NOTE: this method uses the shared parser instance initialized by a previous init() call
087       * @param xmlfile : local pathname or uri of XML file
088       */
089      public static Document toJDOM(String xmlfile) throws JDOMException, IOException {
090    
091         Document jdoc = builder.build( xmlfile );
092         return jdoc;
093    
094      } // toJDOM()
095    
096      /**
097       * Method to parse an input XML String and return a JDOM document
098       * NOTE: this method uses the shared parser instance initialized by a previous init() call
099       * @param xmlfile : XML serialized as String
100       */
101      public static Document StringToJDOM(String xmlstring) throws JDOMException, IOException {
102    
103         StringReader sr = new StringReader(xmlstring);
104         Document jdoc =  builder.build(sr);
105         return jdoc;
106    
107      } // StringToJDOM()
108    
109      /**
110       * Method to parse an input xml file and return a string representation
111       * NOTE: this method uses the shared parser instance initialized by a previous init() call
112       * @param xmlfile : local pathname or uri of XML file
113       */
114      public static String toString(String xmlfile, boolean validate) throws JDOMException, IOException {
115        Document jdoc = toJDOM(xmlfile, validate);
116        XMLOutputter outputter = new XMLOutputter(org.jdom.output.Format.getCompactFormat());
117        return outputter.outputString(jdoc);
118      } // toString()
119    
120      /**
121       * Method to parse an input xml file and return a JDOM document
122       * @param xmlfile : local pathname or uri of XML file
123       * @param validate : true to turn on schema validation
124       */
125      public static Document toJDOM(String xmlfile, boolean validate) throws JDOMException, IOException {
126    
127         SAXBuilder _builder = getBuilder(validate);
128         Document jdoc = _builder.build( xmlfile );
129         return jdoc;
130    
131      } // toJDOM()
132      
133      /**
134       * Method to parse an input xml file retrieved with the class loader and return a JDOM document
135       * @param xmlfile : name of XML file, located in classpath
136       * @param validate : true to turn on schema validation
137       */
138      public static Document classpathToJDOM(String xmlfile, boolean validate) throws JDOMException, IOException {
139    
140         SAXBuilder _builder = getBuilder(validate);
141         if (xmlfile.indexOf("classpath:")>=0) xmlfile = xmlfile.substring(10);  // remove "classpath:"
142         Document jdoc = _builder.build((new Parser()).getClass().getClassLoader().getResourceAsStream(xmlfile));
143         return jdoc;
144    
145      } // classpathToJDOM()
146      
147      
148      /**
149       * Method to load an XML resource found in the classpath and return a JDOM document
150       * @param filename  path of XML resource relative to classpath (starting with a '/'), for example: /somedir/somefile.xml
151       * @param validate  true to turn on schema validation
152       */
153      public static Document classpathResourceToJDOM(String filename, boolean validate) throws JDOMException, IOException {
154         SAXBuilder _builder = getBuilder(validate);     
155         InputStream is = ClassLoader.getSystemClassLoader().getResource(filename).openStream();
156         //InputStream is = new Object().getClass().getResourceAsStream (filename);
157         Document jdoc = _builder.build( is );
158         return jdoc;
159    
160      } // classpathResourceToJDOM()
161      
162    
163      /**
164       * Method to parse an input XML String and return a JDOM document
165       * @param xmlfile : local pathname or uri of XML file
166       * @param validate : true to turn on schema validation
167       */
168      public static Document StringToJDOM(String xmlstring, boolean validate) throws JDOMException, IOException {
169    
170         StringReader sr = new StringReader(xmlstring);
171         SAXBuilder _builder = getBuilder(validate);
172         Document jdoc =  _builder.build(sr);
173         return jdoc;
174    
175      } // StringToJDOM()
176      
177      /** Utility method to load and configure XML parser */
178      private static SAXBuilder getBuilder(boolean validate) {
179            
180            SAXBuilder _builder = new SAXBuilder(parser, validate);
181        _builder.setFeature("http://xml.org/sax/features/namespaces",true);
182        //  builder.setIgnoringElementContentWhitespace(true);
183        
184        // load schema locations
185        if (validate) {
186                    _builder.setFeature("http://apache.org/xml/features/validation/schema",true);
187                ResourceBundle rb = PropertyResourceBundle.getBundle("ucar.xml.schemas");
188                Enumeration schemas = rb.getKeys();
189                StringBuffer sb = new StringBuffer();
190                while (schemas.hasMoreElements()) {
191                            String schema = (String)schemas.nextElement();
192                            String location = rb.getString(schema);
193                            sb.append(sb.length()>0 ? " " : "").append(schema).append(" ").append(location);
194                }
195                _builder.setProperty("http://apache.org/xml/properties/schema/external-schemaLocation", sb.toString());
196     
197                //  cache grammars for subsequent reuse
198                //builder.setFeature("http://apache.org/xml/features/validation/cache-grammarFromParse",true);
199        } // validate
200        
201        return _builder;
202        
203      } // getBuilder()
204    
205    } // Parser
