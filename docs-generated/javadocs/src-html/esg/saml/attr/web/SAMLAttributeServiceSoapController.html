019    package esg.saml.attr.web;
020    
021    import java.io.InputStream;
022    
023    import javax.servlet.http.HttpServletRequest;
024    import javax.servlet.http.HttpServletResponse;
025    
026    import org.springframework.beans.factory.annotation.Autowired;
027    import org.springframework.beans.factory.annotation.Qualifier;
028    import org.springframework.web.bind.annotation.RequestMapping;
029    import org.springframework.web.bind.annotation.RequestMethod;
030    
031    import esg.saml.attr.service.api.SAMLAttributeService;
032    import esg.saml.common.SAMLParameters;
033    
034    /**
035     * Controller endpoint for SAML attribute requests with SOAP binding.
036     * This controller is responsible only for managing the input and output streams of the HTTP request:
037     * all other functionality is delegated to the underlying {@link SAMLAttributeService} specific to SOAP binding.
038     */
039    //@Controller("samlAttributeServiceSoapController")
040    @RequestMapping("/saml/soap/secure/attributeService.htm")
041    //@RequestMapping("/secure/client-cert/saml/soap/attributeService.htm")
042    public class SAMLAttributeServiceSoapController {
043            
044            private final SAMLAttributeService samlAttributeService;
045            
046            @Autowired
047            public SAMLAttributeServiceSoapController(final @Qualifier("samlAttributeService") SAMLAttributeService samlService) {
048                    this.samlAttributeService = samlService;
049            }
050            
051            @RequestMapping(method = { RequestMethod.GET, RequestMethod.POST} )
052            public void process(final HttpServletRequest httpRequest, final HttpServletResponse httpResponse) throws Exception {
053                    
054                    // read SOAP/SAML request from HTTP request
055                    final InputStream inputStream = httpRequest.getInputStream();   
056                    
057                    // process SOAP/SAML request
058                    final String xml = samlAttributeService.processAttributeQuery(inputStream);
059                    
060                    // write SOAP/SAML response to HTTP response
061                    httpResponse.setContentType(SAMLParameters.CONTENT_TYPE_XML);
062                    httpResponse.getWriter().write( xml );
063            
064            }
065    
066    }
