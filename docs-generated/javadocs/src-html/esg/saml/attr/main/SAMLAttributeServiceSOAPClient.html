019    package esg.saml.attr.main;
020    
021    import java.io.File;
022    
023    import org.apache.commons.io.FileUtils;
024    import org.springframework.core.io.ClassPathResource;
025    
026    import esg.saml.common.SOAPServiceClient;
027    
028    /**
029     * Example client to contact a deployed ESG-CET SAML Attribute Service via SOAP binding.
030     * This class is setup to query a SAML Attribute Service deployed on localhost.
031     */
032    public class SAMLAttributeServiceSOAPClient {
033    
034              // use HTTP endpoint for application deployed on localhost
035              private static final String ENDPOINT = "http://localhost:8080/SamlProject/saml/soap/secure/attributeService.htm";       
036              private static final String SAML_REQUEST = "esg/saml/attr/main/SAMLattributeQueryRequest.xml";
037              
038              public static void main(String[] args) throws Exception {
039                      
040                      final File file = new ClassPathResource(SAML_REQUEST).getFile();
041                      final String samlRequest = FileUtils.readFileToString(file);
042                      final SOAPServiceClient client = new SOAPServiceClient();
043                      client.doSoap(ENDPOINT, samlRequest);
044                                              
045              }
046            
047    }
