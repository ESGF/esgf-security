019    package esg.saml.attr.service.api;
020    
021    import org.opensaml.xml.io.MarshallingException;
022    import org.opensaml.xml.io.UnmarshallingException;
023    import org.opensaml.xml.parse.XMLParserException;
024    
025    /**
026     * Client-side API for interacting with a {@link SAMLAttrbuteService}.
027     * Note that this API is binding-agnostic.
028     */
029    public interface SAMLAttributeServiceClient {
030            
031            /**
032             * Method to build a serialized SAML attribute request for a user with given OpenID.
033             * @param openid : the user unique identifier.
034             * @return the SAML attribute request (with binding) serialized as string, to be sent to the SAML attribute service.
035             */
036            String buildAttributeRequest(String identifier) throws MarshallingException;
037    
038            /**
039             * Method to parse a serialized SAML attribute response into a user object.
040             * @param attributeResponse :  the SAML attribute response obtained from the SAML service, serialized as string.
041             * @return : object populated with attributes extracted from the SAML response.
042             */
043            SAMLAttributes parseAttributeResponse(String attributeResponse) throws XMLParserException, UnmarshallingException;
044            
045    }
