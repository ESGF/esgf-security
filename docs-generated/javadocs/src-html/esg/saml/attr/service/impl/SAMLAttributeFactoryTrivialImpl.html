019    package esg.saml.attr.service.impl;
020    
021    import esg.saml.attr.service.api.SAMLAttributeFactory;
022    import esg.saml.attr.service.api.SAMLAttributes;
023    import esg.saml.common.SAMLUnknownPrincipalException;
024    
025    /**
026     * Trivial implementation of {@link SAMLAttributeFactory} that returns attributes for only one user.
027     */
028    public class SAMLAttributeFactoryTrivialImpl implements SAMLAttributeFactory {
029    
030            public SAMLAttributeFactoryTrivialImpl() {}
031            
032            private String issuer;
033    
034            public SAMLAttributes newInstance(final String identifier) throws SAMLUnknownPrincipalException {
035                    
036                    if (identifier.equals("Test Openid")) {
037                            
038                            final SAMLAttributes samlAttributes = new SAMLAttributesImpl();
039                            
040                            // user information
041                            samlAttributes.setFirstName("Test First Name");
042                            samlAttributes.setLastName("Test Last Name");
043                            samlAttributes.setOpenid("Test Openid");
044                            samlAttributes.setEmail("Test Email");
045                            
046                            // access control attributes
047                            samlAttributes.getAttributes().add("group_TestGroup_role_default");
048                            
049                            // authority
050                            samlAttributes.setIssuer(this.getIssuer());
051                            
052                            return samlAttributes;
053                    
054                    } else {
055                            throw new SAMLUnknownPrincipalException("Unknown principal: "+identifier);
056                    }
057                    
058            }
059            
060            public String getIssuer() {
061                    return issuer;
062            }
063    
064            public void setIssuer(final String issuer) {
065                    this.issuer = issuer;
066            }
067    
068    }
