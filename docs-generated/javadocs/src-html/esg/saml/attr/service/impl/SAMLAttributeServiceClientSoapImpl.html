019    package esg.saml.attr.service.impl;
020    
021    import java.util.Collection;
022    
023    import org.apache.commons.logging.Log;
024    import org.apache.commons.logging.LogFactory;
025    import org.opensaml.saml2.core.Assertion;
026    import org.opensaml.saml2.core.AttributeQuery;
027    import org.opensaml.saml2.core.Response;
028    import org.opensaml.ws.soap.soap11.Body;
029    import org.opensaml.ws.soap.soap11.Envelope;
030    import org.opensaml.xml.io.MarshallingException;
031    import org.opensaml.xml.io.UnmarshallingException;
032    import org.opensaml.xml.parse.XMLParserException;
033    import org.opensaml.xml.util.XMLHelper;
034    import org.w3c.dom.Element;
035    import org.w3c.dom.Node;
036    
037    import esg.saml.attr.service.api.SAMLAttributeServiceClient;
038    import esg.saml.attr.service.api.SAMLAttributeStatementHandler;
039    import esg.saml.attr.service.api.SAMLAttributes;
040    import esg.saml.common.SAMLBuilder;
041    
042    /**
043     * Implementation of {@link SAMLAttributeServiceClient} specific to SOAP binding.
044     */
045    public class SAMLAttributeServiceClientSoapImpl implements SAMLAttributeServiceClient {
046            
047            private final SAMLAttributeQueryRequestBuilderImpl requestBuilder;
048            private final SAMLAttributeStatementHandler statementHandler;
049            private final SAMLBuilder samlBuilder;
050            
051            private String issuer;
052            
053            private final static Log LOG = LogFactory.getLog(SAMLAttributeServiceClientSoapImpl.class);
054            
055            public SAMLAttributeServiceClientSoapImpl(final String issuer) {
056                    
057                    this.issuer = issuer;
058                                    
059                    // instantiate SAMLBuilder
060                    this.samlBuilder = SAMLBuilder.getInstance();
061                    
062                    // instantiate SAML handlers
063                    this.requestBuilder = new SAMLAttributeQueryRequestBuilderImpl();       
064                    this.statementHandler = new SAMLAttributeStatementHandlerImpl();
065                    
066            }
067    
068            /**
069             * {@inheritDoc}
070             */
071            public String buildAttributeRequest(final String openid) throws MarshallingException {
072                    
073                    // build attribute query
074                    final AttributeQuery attributeQuery = requestBuilder.buildAttributeQueryRequest(openid, issuer);
075                    
076                    // embed into SOAP envelop
077                    final Envelope soapRequestEnvelope = samlBuilder.getSOAPEnvelope();
078                    final Body soapRequestBody = samlBuilder.getSOAPBody();
079                    soapRequestBody.getUnknownXMLObjects().add(attributeQuery);
080                    soapRequestEnvelope.setBody(soapRequestBody);
081                    
082                    // serialize
083                    final Element soapRequestElement = samlBuilder.marshall(soapRequestEnvelope);
084                    final String xml = XMLHelper.prettyPrintXML((Node)soapRequestElement);
085                    if (LOG.isDebugEnabled()) LOG.debug("SOAP request:\n"+xml);
086                    
087                    return xml;
088    
089            }
090    
091            /**
092             * {@inheritDoc}
093             */
094            public SAMLAttributes parseAttributeResponse(final String attributeResponse) throws XMLParserException, UnmarshallingException {
095                    
096                    // empty attributes by default
097                    SAMLAttributes attributes = new SAMLAttributesImpl();
098                    if (LOG.isDebugEnabled()) LOG.debug("Parsing attribute response=\n"+attributeResponse);
099                    
100                    // string > DOM
101            final Element soapResponseElement = samlBuilder.parse(attributeResponse);
102            
103            // DOM > SAML objects
104            final Envelope soapResponseEnvelope = (Envelope)samlBuilder.unmarshall(soapResponseElement);
105            final Body soapResponseBody = soapResponseEnvelope.getBody();
106            final Response samlResponse = (Response)soapResponseBody.getUnknownXMLObjects().get(0);
107            
108            // SAML object > User object
109            final Collection<Assertion> assertions = samlResponse.getAssertions();
110            for (final Assertion assertion : assertions) {
111                    attributes = statementHandler.parseAttributeStatement(assertion);
112            }
113    
114            return attributes;
115            }
116            
117    }
