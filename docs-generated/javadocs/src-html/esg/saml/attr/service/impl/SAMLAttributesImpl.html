019    package esg.saml.attr.service.impl;
020    
021    import java.util.Set;
022    import java.util.TreeSet;
023    
024    import esg.saml.attr.service.api.SAMLAttributes;
025    
026    /**
027     * Bean implementation of the {@link SAMLAttributes} interface.
028     */
029    public class SAMLAttributesImpl implements SAMLAttributes {
030            
031            private String firstName;
032            
033            private String lastName;
034            
035            private String openid;
036            
037            private String email;
038            
039            public SAMLAttributesImpl() {}
040            
041            /**
042             * The authority that issued these attributes.
043             */
044            private String issuer;
045            
046            /**
047             * Note that attributes are naturally ordered.
048             */
049            private Set<String> attributes = new TreeSet<String>();
050    
051            public String getFirstName() {
052                    return firstName;
053            }
054    
055            public void setFirstName(String firstName) {
056                    this.firstName = firstName;
057            }
058    
059            public String getLastName() {
060                    return lastName;
061            }
062    
063            public void setLastName(String lastName) {
064                    this.lastName = lastName;
065            }
066    
067            public String getOpenid() {
068                    return openid;
069            }
070    
071            public void setOpenid(String openid) {
072                    this.openid = openid;
073            }
074    
075            public String getEmail() {
076                    return email;
077            }
078    
079            public void setEmail(String email) {
080                    this.email = email;
081            }
082    
083            public Set<String> getAttributes() {
084                    return attributes;
085            }
086    
087            public void setAttributes(Set<String> attributes) {
088                    this.attributes = attributes;
089            }
090    
091            public String getIssuer() {
092                    return issuer;
093            }
094    
095            public void setIssuer(String issuer) {
096                    this.issuer = issuer;
097            }
098            
099    }
