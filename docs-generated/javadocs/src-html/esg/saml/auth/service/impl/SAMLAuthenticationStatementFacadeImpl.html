019    package esg.saml.auth.service.impl;
020    
021    import java.io.File;
022    import java.util.HashMap;
023    import java.util.Map;
024    
025    import org.apache.commons.logging.Log;
026    import org.apache.commons.logging.LogFactory;
027    import org.opensaml.saml2.core.Assertion;
028    import org.opensaml.xml.io.UnmarshallingException;
029    import org.opensaml.xml.parse.XMLParserException;
030    import org.opensaml.xml.security.SecurityException;
031    import org.opensaml.xml.security.credential.Credential;
032    import org.springframework.util.Assert;
033    import org.w3c.dom.Element;
034    
035    import esg.saml.auth.service.api.SAMLAuthenticationStatementFacade;
036    import esg.saml.common.SAMLBuilder;
037    import esg.saml.common.SAMLInvalidStatementException;
038    import esg.saml.common.SecurityManager;
039    import esg.utils.xml.Serializer;
040    
041    /**
042     * Implementation of {@link SAMLAuthenticationStatementFacade} that stores the optional signing and trusted credentials.
043     *
044     */
045    public class SAMLAuthenticationStatementFacadeImpl implements SAMLAuthenticationStatementFacade {
046            
047            private final SAMLBuilder builder;
048            private SAMLAuthenticationStatementHandlerImpl samlAuthenticationStatementHandler;
049            
050            /**
051             * Optional credential to sign SAML statements.
052             */
053            private Credential signingCredential;
054            
055            /**
056             * Optional set of trusted credentials to validate signed SAML statements.
057             */
058            private Map<String,Credential> trustedCredentials = new HashMap<String, Credential>();
059            
060            protected final static Log LOG = LogFactory.getLog(SAMLAuthenticationStatementFacadeImpl.class);
061            
062            /**
063             * Minimal constructor.
064             */
065            public SAMLAuthenticationStatementFacadeImpl() {
066                    
067                    this.builder = SAMLBuilder.getInstance();
068                    this.samlAuthenticationStatementHandler = new SAMLAuthenticationStatementHandlerImpl();
069                    
070            }
071            
072            /**
073             * {@inheritDoc}
074             */
075            public String buildAuthenticationStatement(final String identity, final String issuer, final boolean signit) throws Exception {
076                    
077                    // build statement
078                    final Assertion assertion = samlAuthenticationStatementHandler.buildAuthenticationStatement(identity, issuer);
079                    
080                    // optionally sign statement
081                    if (signit) {
082                            Assert.notNull(signingCredential);
083                            final Element assertionElement = builder.marshallAndSign(assertion, signingCredential);
084                            return Serializer.DOMtoString(assertionElement);
085                    } else {
086                            final Element assertionElement = builder.marshall(assertion);
087                            return Serializer.DOMtoString(assertionElement);
088                    }
089    
090            }
091    
092            /**
093             * {@inheritDoc}
094             */
095            public String buildSignedAuthenticationStatement(final String identity) throws Exception {
096                    
097                    return this.buildAuthenticationStatement(identity, signingCredential.getEntityId(), true);
098                    
099            }
100            
101            /**
102             * {@inheritDoc}
103             */
104            public String parseAuthenticationStatement(final String xml, final boolean validate) throws SAMLInvalidStatementException {
105                    
106                    try {
107                            
108                            final Element assertionElement = builder.parse(xml);
109                    final Assertion assertion = (Assertion)builder.unmarshall(assertionElement); 
110                    if (validate) this.validateAssertion(assertion);        
111                    return samlAuthenticationStatementHandler.parseAuthenticationStatement(assertion);
112            
113                    } catch(XMLParserException e) {
114                            throw new SAMLInvalidStatementException(e);
115                    } catch(UnmarshallingException e) {
116                            throw new SAMLInvalidStatementException(e);
117                    }
118                    
119            }
120    
121            /**
122             * {@inheritDoc}
123             */
124            public void setSigningCredential(final File keystore, final String keystorePassword, final String keystoreAlias) throws Exception {
125                    
126                    signingCredential = SecurityManager.getMyCredential(keystore, keystorePassword, keystoreAlias);
127                    if (LOG.isDebugEnabled()) LOG.debug("Set signinig credential "+signingCredential.getEntityId());
128                    
129            }
130    
131            /**
132             * {@inheritDoc}
133             */
134            public void setTrustedCredentials(final File trustore, final String trustorePassword) throws Exception {
135                    
136                    trustedCredentials = SecurityManager.getTrustedCredentials(trustore, trustorePassword);
137                    
138            }
139            
140            /**
141             * Utility method to validate a SAML assertion versus the current trustore.
142             * @param assertion
143             * @throws SAMLInvalidStatementException
144             */
145            private void validateAssertion(final Assertion assertion) throws SAMLInvalidStatementException {
146                            
147                    try {
148                            if (!builder.validateSignature(assertion, trustedCredentials)) {
149                                    throw new SAMLInvalidStatementException("SAML statement signature is invalid");
150                            }
151                    } catch(SecurityException e) {
152                            throw new SAMLInvalidStatementException(e);
153                    }
154    
155            }
156    
157            /**
158             * Method to disable inclusion of dates in statements for testing.
159             * @param includeFlag
160             */
161            void setIncludeFlag(final boolean includeFlag) {
162                    this.samlAuthenticationStatementHandler.setIncludeFlag(includeFlag);
163            }
164            
165    }
