019    package esg.saml.auth.service.api;
020    
021    import java.io.File;
022    
023    import esg.saml.common.SAMLInvalidStatementException;
024    
025    /**
026     * High level API to build and process SAML authentication statements.
027     */
028    public interface SAMLAuthenticationStatementFacade {
029            
030            /**
031             * Method to generate a SAML authentication statement.
032             * @param identity : the identity asserted in the SAML statement.
033             * @param issuer : the issuer of the SAML statement.
034             * @param signit : true to digitally sign the statement (requires signing credential).
035             * @return : the SAML authentication statement serialized as a string.
036             */
037            String buildAuthenticationStatement(String identity, String issuer, boolean signit) throws Exception;
038            
039            /**
040             * Shortcut method to generate a signed SAML authentication statement, issued by the identity of the signing credential.
041             * @param identity
042             * @return
043             * @throws Exceoption
044             */
045            String buildSignedAuthenticationStatement(String identity) throws Exception;
046            
047            /**
048             * Method to process a SAML authentication statement and extracted the asserted identity.
049             * @param xml : the SAML authentication statement serialized as a string.
050             * @param validateSignature : true to validate the statement signature (requires a set of trusted credentials).
051             * @return : the asserted identity.
052             * @throws SAMLInvalidStatementException : if the SAML statement did not validate.
053             */
054            String parseAuthenticationStatement(String xml, boolean validateSignature) throws SAMLInvalidStatementException;
055            
056            /**
057             * Method to specify the optional credential to sign SAML assertions.
058             * @param keystore
059             * @param keystorePassword
060             * @param keystoreAlias
061             */
062            void setSigningCredential(File keystoreFile, String keystorePassword, String keystoreAlias) throws Exception;
063            
064            /**
065             * Method to specify a set of optional trusted credentials to validate the statement signature.
066             * @param trustore
067             * @param trustorePassword
068             */
069            void setTrustedCredentials(File trustoreFile, String trustorePassword) throws Exception;
070    
071    }
