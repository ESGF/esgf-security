019    package esg.saml.authz.service.impl;
020    
021    import java.util.Vector;
022    
023    import esg.saml.authz.service.api.SAMLAuthorization;
024    import esg.saml.authz.service.api.SAMLAuthorizations;
025    
026    /**
027     * Bean implementation of the {@link SAMLAuthorizations} interface.
028     */
029    public class SAMLAuthorizationsImpl implements SAMLAuthorizations {
030            
031            private Vector<SAMLAuthorization> authorizations = null;
032            
033            /**
034             * The authority that issued these attributes.
035             */
036            private String issuer;
037            
038            /**
039             * The identity about which the authorization statements are made.
040             */
041            private String identity;
042    
043            public SAMLAuthorizationsImpl() {
044                    this.authorizations = new Vector<SAMLAuthorization>();
045            }
046    
047            public Vector<SAMLAuthorization> getAuthorizations() {
048                    return this.authorizations;
049            }
050    
051            public void addAuthorization(final SAMLAuthorization authz) {
052                    this.authorizations.add(authz);
053            }
054    
055            public String getIssuer() {
056                    return issuer;
057            }
058    
059            public void setIssuer(String issuer) {
060                    this.issuer = issuer;
061            }
062    
063            public String getIdentity() {
064                    return identity;
065            }
066    
067            public void setIdentity(String identity) {
068                    this.identity = identity;
069            }
070            
071    }
