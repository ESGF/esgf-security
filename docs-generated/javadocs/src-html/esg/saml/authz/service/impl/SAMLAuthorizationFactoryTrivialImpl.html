019    package esg.saml.authz.service.impl;
020    
021    import java.util.Vector;
022    
023    import org.opensaml.saml2.core.DecisionTypeEnumeration;
024    
025    import esg.saml.authz.service.api.SAMLAuthorization;
026    import esg.saml.authz.service.api.SAMLAuthorizationFactory;
027    import esg.saml.authz.service.api.SAMLAuthorizations;
028    import esg.saml.common.SAMLUnknownPrincipalException;
029    
030    /**
031     * Trivial implementation of {@link SAMLAuthorizationFactory} that returns positive authorization
032     * for a given user (for all resources and actions), negative authorization otherwise.
033     */
034    public class SAMLAuthorizationFactoryTrivialImpl implements SAMLAuthorizationFactory {
035    
036            public SAMLAuthorizationFactoryTrivialImpl() {}
037            
038            private String issuer;
039    
040            public SAMLAuthorizations newInstance(final String identifier, final String resource, final Vector<String> actions) throws SAMLUnknownPrincipalException {
041                            
042                    if (identifier.equals("Test Openid")) {
043                            
044                            final SAMLAuthorization samlAuthorization = new SAMLAuthorizationImpl();                
045                            samlAuthorization.setResource(resource);
046                            samlAuthorization.setActions(actions);
047                            samlAuthorization.setDecision(DecisionTypeEnumeration.PERMIT.toString());
048                            
049                            final SAMLAuthorizations samlAuthorizations = new SAMLAuthorizationsImpl();
050                            samlAuthorizations.setIdentity(identifier);
051                            if (issuer!=null) samlAuthorizations.setIssuer(issuer);
052                            samlAuthorizations.addAuthorization(samlAuthorization);
053                            
054                            return samlAuthorizations;
055                    
056                    } else {
057                            throw new SAMLUnknownPrincipalException("Unknown user: "+identifier);
058                    }
059            
060            }
061            
062            public String getIssuer() {
063                    return issuer;
064            }
065    
066            public void setIssuer(final String issuer) {
067                    this.issuer = issuer;
068            }
069    
070    }
