019    package esg.saml.authz.service.api;
020    
021    import org.opensaml.xml.io.MarshallingException;
022    import org.opensaml.xml.io.UnmarshallingException;
023    import org.opensaml.xml.parse.XMLParserException;
024    
025    /**
026     * Client-side API for interacting with a {@link SAMLAuthorizationService}.
027     * Note that this API is binding-agnostic.
028     */
029    public interface SAMLAuthorizationServiceClient {
030                    
031            /**
032             * Method to build a serialized SAML authorization request for a user with given OpenID, a given resource, and a given action.
033             * @param openid : the user unique identifier.
034             * @param resource : the resource requested.
035             * @param action : the operation to be performed on the resource.
036             * @return the SAML authorization request (with binding) serialized as string, to be sent to the SAML attribute service.
037             */
038            String buildAuthorizationRequest(String openid, String resource, String action) throws MarshallingException;
039    
040            /**
041             * Method to parse a serialized SAML authorization response into a user object.
042             * @param authorizationResponse :  the SAML authorization response obtained from the SAML service, serialized as string.
043             * @return : object populated with authorizations extracted from the SAML response.
044             */
045            SAMLAuthorizations parseAuthorizationResponse(String authorizationResponse) throws XMLParserException, UnmarshallingException;
046    }
